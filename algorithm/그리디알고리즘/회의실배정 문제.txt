#define _CRT_SECURE_NO_DEPRECATE
#include<iostream>
#include<cstdio>
#include<algorithm>
#include<vector>
#include<utility>

using namespace std;

int main(void) {

	int n,x,y,sum=0;
	//ют╥б
	scanf("%d", &n);

	vector< pair<int, int> > arr;

	for (int i = 0; i < n; i++) {
		cin >> x >> y;
		arr.push_back(make_pair(y, x));
	}
	
	sort(arr.begin(), arr.end());

	int end = -1;

	for (int i = 0; i < arr.size(); i++)
	{
		if (arr[i].second >= end) 
		{
			end = arr[i].first;
			sum++;
		}
	}
	
	cout << sum << endl;

	return 0;
}
